בִּרְכַּת הַמָּזוֹן

נֹסַח אֶרֶץ יִשְׂרָאֵל

בִּרְכַּת הַזָּן

בָּרוּךְ אַתָּה, יי, אֱלֹהֵינוּ, מֶלֶךְ הָעוֹלָם, הַזָּן אוֹתָנוּ וְאֶת הָעוֹלָם כֻּלּוֹ, בְּטוֹב, בְּחֵן, בְּחֶסֶד וּבְרַחֲמִים. רְעֵנוּ, פַּרְנְסֵנוּ, וְאֶל תַּצְרִיכֵנוּ לִידֵי מַתְּנַת בָּשָׂר וָדָם, שֶׁלֹּא נֵבוֹשׁ וְלֹא נִכָּלֵם לְעוֹלָם וָעֶד. כָּאָמוּר: פּוֹתֵחַ אֶת יָדֶךָ, וּמַשְׂבִּיעַ לְכָל חַי רָצוֹן. בָּרוּךְ אַתָּה, יי, הַזָּן אֶת הַכֹּל.

בִּרְכַּת הָאָרֶץ

נוֹדֶה לְךָ, יי, אֱלֹהֵינוּ. כִּי הִנְחַלְתָּנוּ אֶרֶץ חֶמְדָּה טוֹבָה וּרְחָבָה, בְּרִית וְתוֹרָה, חַיִּים וּמָזוֹן.

עַל הַנִּסִּים

עַל הַנִּסִּים, הַגְּבוּרוֹת, הַתְּשוּעוֹת, הַמִּלְחָמוֹת וְהַפְּדוּת שֶׁעָשִׂיתָ עִמָּנוּ וְעִם אֲבוֹתֵינוּ בַּיָּמִים הָהֵם בַּזְּמַן הַזֶּה.

לַחֲנֻכָּה:

בִּימֵי מַתִּתְיָהוּ בֶּן יוֹחָנָן כֹּהֵן גָּדוֹל חַשְׁמוֹנַאי וּבָנָיו, כְּשֶׁעָמְדָה מַלְכוּת יָוָן הָרְשָׁעָה עַל עַמְּךָ יִשְׂרָאֵל לְהַשְׁכִּיחָם תּוֹרָתֶךָ וּלְהַעֲבִירָם מֵחֻקֵּי רְצוֹנֶךָ, וְאַתָּה בְּרַחֲמֶיךָ הָרַבִּים עָמַדְתָּ לָהֶם בְּעֵת צָרָתָם, רַבְתָּ אֶת רִיבָם, דַּנְתָּ אֶת דִּינָם, נָקַמְתָּ אֶת נִקְמָתָם, מָסַרְתָּ גִּבּוֹרִים בְּיַד חַלָּשִׁים וְרַבִּים בְּיַד מְעַטִּים וּרְשָׁעִים בְּיַד צַדִּיקִים וּטְמֵאִים בְּיַד טְהוֹרִים וְזֵדִים בְּיַד עוֹסְקֵי תוֹרָתֶךָ, וּלְךָ עָשִׂיתָ שֵׁם גָּדוֹל וְקָדוֹשׁ בְּעוֹלָמֶךָ וּלְעַמְּךָ יִשְׂרָאֵל עָשִׂיתָ תְּשׁוּעָה גְדוֹלָה וּפֻרְקָן כְּהַיּוֹם הַזֶּה, וְאַחַר כָּךְ בָּאוּ בָנֶיךָ לִדְבִיר בֵּיתֶךָ וּפִנּוּ אֶת הֵיכָלֶךָ וְטִהֲרוּ אֶת מִקְדָּשֶׁךָ וְהִדְלִיקוּ נֵרוֹת בְּחַצְרוֹת קָדְשֶׁךָ וְקָבְעוּ שְׁמוֹנַת יְמֵי חֲנֻכָּה אֵלֶּה בְּהַלֵּל וּבְהוֹדָאָה.

לְפּוּרִים:

בִּימֵי מָרְדֳּכַי וְאֶסְתֵּר בְּשׁוּשַׁן הַבִּירָה, כְּשֶׁעָמַד עֲלֵיהֶם הָמָן הָרָשָׁע, בִּקֵּשׁ לְהַשְׁמִיד לַהֲרֹג וּלְאַבֵּד אֶת כָּל הַיְּהוּדִים מִנַּעַר וְעַד זָקֵן טַף וְנָשִׁים בְּיוֹם אֶחָד, בִּשְׁלוֹשָׁה עָשָׂר לְחֹדֶשׁ שְׁנֵים עָשָׂר הוּא חֹדֶשׁ אֲדָר וּשְׁלָלָם לָבוֹז. וְאַתָּה בְּרַחֲמֶיךָ הָרַבִּים הֵפַרְתָּ אֶת עֲצָתוֹ וְקִלְקַלְתָּ אֶת מַחֲשַׁבְתּוֹ וַהֲשֵׁבוֹתָ לוֹ גְּמוּלוֹ בְּרֹאשׁוֹ וְתָלוּ אוֹתוֹ וְאֶת בָּנָיו עַל הָעֵץ.

לְיוֹם הָעַצְמָאוּת:

בִּימֵי תְקוּמָתֵנוּ, בְּקוּם עָלֵינוּ בְּנֵי עֲרָב, לְהַשְׁמִיד לַהֲרֹג וּלְאַבֵּד אֶת הָעֹלִים מִשְּׁבִי הַגּוֹלָה אֶל אֶרֶץ חֶמְדָּה, אָמְרוּ: 'לְכוּ וְנַכְחִידֵם מִגּוֹי וְלֹא יִזָּכֵר שֵׁם יִשְׂרָאֵל עוֹד'. וְאַתָּה בְּרַחֲמֶיךָ הָרַבִּים עָמַדְתָּ לָּנוּ בְּעֵת צָרָתֵנוּ, רַבְתָּ אֶת רִיבֵנוּ, דַּנְתָּ אֶת דִּינֵנוּ, נָקַמְתָּ אֶת נִקְמָתֵנוּ, מָסַרְתָּ רַבִּים בְּיַד מְעַטִּים וּרְשָׁעִים בְּיַד צַדִּיקִים, 'הֵמָּה כָּרְעוּ וְנָפָלוּ וַאֲנַחְנוּ קַּמְנוּ וַנִּתְעוֹדָד'. לְךָ עָשִׂיתָ שֵׁם גָּדוֹל בָּעוֹלָם וּלְעַמְּךָ יִשְׂרָאֵל עָשִׂיתָ תְּשׁוּעָה גְדוֹלָה, וּבַחֹדֶשׁ הַשֵּׁנִי בַּחֲמִשָּׁה לַחֹדֶשׁ, פָּרַקְנוּ עוֹל גּוֹיִם מֵעַל צַוָּארֵנוּ.

לְיוֹם גְּאֻלַּת יְרוּשָׁלַיִם:

בִּשְׁנַת עֶשְׂרִים לִתְקוּמָתֵנוּ, בְּקוּם עָלֵינוּ בְּנֵי עֲרָב. אָמְרוּ: לְכוּ וְנַכְחִידֵם מִגּוֹי, וְלֹא יִזָּכֵר שֵׁם יִשְׂרָאֵל עוֹד. וְאַתָּה בְּרַחֲמֶיךָ הָרַבִּים עָמַדְתָּ לָּנוּ בְּעֵת צָרָתֵנוּ. רַבְתָּ אֶת רִיבֵנוּ, דַּנְתָּ אֶת דִּינֵנוּ, נָקַמְתָּ אֶת נִקְמָתֵנוּ. מָסַרְתָּ רַבִּים בְּיַד מְעַטִּים, וּרְשָׁעִים בְּיַד צַדִּיקִים. וַאֲנַחְנוּ קַּמְנוּ וַנִּתְעוֹדָד. בָּאוּ צִבְאוֹתֶיךָ אֶל חַבְלֵי אֶרֶץ נַחֲלָה אֲשֶׁר נִשְׁבַּעְתָּ לַאֲבֹתֵינוּ לָתֶת לָנוּ. לְךָ עָשִׂיתָ שֵׁם גָּדוֹל בָּעוֹלָם, וּלְעַמְּךָ יִשְׂרָאֵל עָשִׂיתָ תְּשׁוּעָה גְדוֹלָה. וּבַחֹדֶשׁ הַשֵּׁנִי בִּשְׁמוֹנָה וְעֶשְׂרִים בּוֹ, עֹמְדוֹת הָיוּ רַגְלֵינוּ בִּשְׁעָרַיִךְ יְרוּשָׁלָיִם.

סיום: כְּשֵׁם שֶׁעָשִׂיתָ לָּנוּ תְּשׁוּעָה בַּיָּמִים הָהֵם, כָּךְ עֲשֵׂה עִמָּנוּ בָּעֵת הַזֹּאת, וְנוֹדֶה לְשִׁמְךָ לָנֶצַח.

וְעַל כֻּלָּם אָנוּ מוֹדִים לְךָ וּמְבָרְכִים אֶת שִׁמְךָ לְעוֹלָם וָעֶד, כַּכָּתוּב: וְאָכַלְתָּ, וְשָׂבַעְתָּ; וּבֵרַכְתָּ אֶת יי אֱלֹהֶיךָ, עַל הָאָרֶץ הַטֹּבָה אֲשֶׁר נָתַן לָךְ.

בָּרוּךְ אַתָּה, יי, על הָאָרֶץ וְעַל הַמָּזוֹן.

בּוֹנֵה יְרוּשָׁלָיִם

רַחֵם, יי, אֱלֹהֵינוּ, בְּרַחֲמֶיךָ הָרַבִּים עַל יִשְׂרָאֵל עַמֶּךָ, עַל יְרוּשָׁלַיִם עִירֶךָ וְעַל צִיּוֹן מִשְׁכַּן כְּבוֹדֶךָ. הָאֵר פָּנֶיךָ עַל מִקְדָּשְׁךָ הַשָּׁמֵם. בְּנֵה בֵּיתְךָ, שַׁכְלֵל הֵיכָלֶךָ. וּמַלְכוּת בֵּית דָּוִד הַחֲזֵר לִמְקוֹמָהּ בְּיָמֵינוּ.

יַעֲלֶה וְיָבֹא

אָנָּא, אֱלֹהֵינוּ וֶאֱלֹהֵי אֲבוֹתֵינוּ, יַעֲלֶה וְיָבוֹא, יַגִּיעַ וְיִרְאֶה, יִזָּכֵר וְיִפָּקֵד לְפָנֶיךָ זִכְרוֹנֵנוּ וְזִכְרוֹן אֲבוֹתֵינוּ, זִכְרוֹן כָּל עַמְּךָ בֵּית יִשְׂרָאֵל וְזִכְרוֹן עִירְךָ וּמִקָּדְשְׁךָ לְטוֹבָה בְּיוֹם רֹאשׁ הַחֹדֶשׁ הַזֶּה, לְרַחֵם עָלֵינוּ בוֹ וּלְהוֹשִׁיעֵנוּ, כִּי אֶל וְרַחוּם אַתָּה לְבַדְּךָ נִקְרֵאתָ. אָז תַּחְפֹּץ זִבְחֵי צֶדֶק, עוֹלָה וְכָלִיל, אָז יַעֲלוּ עַל מִזְבַּחֲךָ פָרִים. וְהַשִּׂיאֶנּוּ, יי אֱלֹהֵינו אֶת בִּרְכַּת מוֹעֲדֶיךָ לְשָׁלוֹם, כַּאֲשֶׁר אָמַרְתָּ וְרָצִיתָ, כֵּן תְּבָרְכֶנּוּ וּבְנֵה אֶת יְרוּשָׁלַיִם בְּקָרוֹב כַּאֲשֶׁר דִּבַּרְתָּ.

בָּרוּךְ אַתָּה, יי, בּוֹנֶה יְרוּשָׁלַיִם. אָמֵן.

הַטּוֹב וְהַמֵּטִיב

בָּרוּךְ אַתָּה, יי, אֱלֹהֵינוּ, מֶלֶךְ הָעוֹלָם, הָאֵל, אָבִינוּ, מַלְכֵּנוּ, בּוֹרְאֵנוּ, גּוֹאֲלֵנוּ, הַמֶּלֶךְ הַטּוֹב וְהַמֵּטִיב, שֶׁבְּכָל יוֹם וָיוֹם הוּא מַרְבֶּה לְהֵטִיב עִמָּנוּ, וְהוּא יִגְמְלֵנוּ לָעַד חֵן וָחֶסֶד, רֶוַח וְרַחֲמִים וְכָל טוֹב.

ברכת האורח

הָרַחֲמָן יְבָרֵךְ אֶת בַּעַל הַבַּיִת הַזֶּה וּבַעַל הַסְּעוּדָה הַזֹּאת בְּבָנִים וּבִנְכָסִים, בְּעֹשֶׁר וּבְכָבוֹד. בָּרֵךְ יי חֵילוֹ וּפֹעַל יָדָיו תִּרְצֶה; מְחַץ מׇתְנַיִם קָמָיו וּמְשַׂנְאָיו מִן יְקוּמוּן. לֹא יֵבוֹשׁ בָּעוֹלָם הַזֶּה, וְלֹא יִכָּלֵם לְעוֹלָם הַבָּא.

הָרַחֲמָן

הָרַחֲמָן יָאִיר עֵינֵינוּ בִּמְאוֹר תּוֹרָה.

יַצְלִיחַ לָנוּ בְּכָל דְּרָכֵינוּ.

יָשִׂים יִרְאָתוֹ לְנֶגֶד עֵינֵינוּ.

יָסִיר מַחֲלָה מִקִּרְבֵּנוּ.

יִגְעַר בְּחוֹשְׁבֵי רָעָתֵנוּ.

הָרַחֲמָן הוּא יְחַדֵּשׁ עָלֵינוּ אֶת הַחֹדֶשׁ הַזֶּה לְטוֹבָה וְלִבְרָכָה.

הָרַחֲמָן יַנְחִילֵנוּ נִצָּחוֹן עַל כָּל אוֹיְבֵינוּ וְיִירַשׁ זַרְעֵנוּ אֵת שַׁעַר שֹׂנְאֵינוּ. יִהְיֶה לָנוּ עוֹזֵר וְסוֹמֵך.

יי עֹז לְעַמּוֹ יִתֵּן יי יְבָרֵךְ אֶת עַמּוֹ בַשָּׁלוֹם
